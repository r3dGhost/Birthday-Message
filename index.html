// Balloons animation function createBalloons() { for (let i = 0; i < 5; i++) { const balloon = document.createElement('div'); balloon.classList.add('balloon'); balloon.style.left = Math.random() * 100 + 'vw'; balloon.style.animationDuration = (5 + Math.random() * 5) + 's'; // Randomize: 50% photo, 50% colorful if (Math.random() > 0.5) { balloon.style.background = `url('trainer.webp') no-repeat center / cover`; balloon.style.borderRadius = "50%"; } else { balloon.style.background = `hsl(${Math.random() * 360}, 70%, 60%)`; } document.body.appendChild(balloon); setTimeout(() => balloon.remove(), 10000); } }
